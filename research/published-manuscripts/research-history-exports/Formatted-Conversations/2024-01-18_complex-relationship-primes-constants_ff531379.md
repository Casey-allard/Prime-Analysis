# Purpose: Implement the Sieve of Eratosthenes and Trial Division for prime number generation and checking

def sieve_of_eratosthenes(limit):
    """ Generate all prime numbers up to the specified limit using the Sieve of Eratosthenes algorithm. """
    if limit < 2:
        return []

    prime_flags = [True] * (limit + 1)
    prime_flags[0] = prime_flags[1] = False  # 0 and 1 are not prime numbers
    for p in range(2, int(limit**0.5) + 1):
        if prime_flags[p]:
            for i in range(p*p, limit + 1, p):
                prime_flags[i] = False

    return [p for p, is_prime in enumerate(prime_flags) if is_prime]

def is_prime_trial_division(number):
    """ Check if a number is prime using the trial division method. """
    if number < 2:
        return False
    for i in range(2, int(number**0.5) + 1):
        if number % i == 0:
            return False
    return True

# Testing the implementations

# Purpose: Analyze prime gaps and perform statistical analysis comparing primes with π and e

def calculate_prime_gaps(primes):
    """ Calculate the gaps between successive prime numbers. """
    return [primes[i] - primes[i-1] for i in range(1, len(primes))]

def perform_statistical_analysis(prime_gaps, pi_digits, e_digits):
    """ Perform statistical analysis on prime gaps and compare with sequences derived from π and e. """
    # Analyzing the frequency of prime gaps
    gap_frequency = {gap: prime_gaps.count(gap) for gap in set(prime_gaps)}

    # Comparing with π and e digits
    pi_gap_correlation = np.corrcoef(prime_gaps, pi_digits[:len(prime_gaps)])[0, 1]
    e_gap_correlation = np.corrcoef(prime_gaps, e_digits[:len(prime_gaps)])[0, 1]

    return gap_frequency, pi_gap_correlation, e_gap_correlation
