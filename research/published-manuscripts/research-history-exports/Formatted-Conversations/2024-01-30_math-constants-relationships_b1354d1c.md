def fractal_dimension(Z, threshold=0.9):
    # Only for 2d image
    assert(len(Z.shape) == 2)

    # From 1x1 to NxN squares
    sizes = np.arange(1, Z.shape[0] + 1)
    Ns = []
    
    # Loop over several scales
    for size in sizes:
        # Number of boxes
        count = 0
        # Loop over the image
        for x in range(0, Z.shape[0], size):
            for y in range(0, Z.shape[1], size):
                if np.sum(Z[x:x + size, y:y + size]) > threshold:
                    count += 1

        Ns.append(count)

    # Fit the relationship between the box count and the box size
    coeffs = linregress(np.log(sizes), np.log(Ns))
